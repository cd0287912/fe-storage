### [js] 写一个方法把下划线命名转成大驼峰命名?

```js
function toCamel(str) {
  if (typeof str !== "string") {
    throw Error("type error");
  }
  const arr = str.split("_");
  if (arr.length > 1) {
    arr.forEach((element, i) => {
      if (i !== 0) {
        arr[i] = element[0].toUpperCase() + element.substring(1, -1);
      }
    });
    return arr.join("");
  }
  return arr[0];
}
```
